67template<typename Float_>
68Float_ find_quantile(Float_ quantile, size_t n, Float_* ptr) {
69 double raw = static_cast<double>(n - 1) * quantile;
70 size_t index = std::ceil(raw);
71 std::nth_element(ptr, ptr + index, ptr + n);
72 double upper = *(ptr + index);
73 std::nth_element(ptr, ptr + index - 1, ptr + index);
74 double lower = *(ptr + index - 1);
75 return lower * (index - raw) + upper * (raw - (index - 1));
76}
