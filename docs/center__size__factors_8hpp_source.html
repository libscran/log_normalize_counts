85template<typename SizeFactor_>
86SizeFactor_ compute_mean(size_t num, const SizeFactor_* size_factors, sanitize_size_factors::Diagnostics* diagnostics, const Options& options) {
87 static_assert(std::is_floating_point<SizeFactor_>::value);
88 SizeFactor_ mean = 0;
89 size_t denom = 0;
90
91 if (options.ignore_invalid) {
92 sanitize_size_factors::Diagnostics tmpdiag;
93 auto& diag = (diagnostics == NULL ? tmpdiag : *diagnostics);
94 for (size_t i = 0; i < num; ++i) {
95 auto val = size_factors[i];
96 if (!sanitize_size_factors::internal::is_invalid(val, diag)) {
97 mean += val;
98 ++denom;
99 }
100 }
101 } else {
102 mean = std::accumulate(size_factors, size_factors + num, static_cast<SizeFactor_>(0));
103 denom = num;
104 }
105
106 if (denom) {
107 return mean/denom;
108 } else {
109 return 0;
110 }
111}

152template<typename SizeFactor_, typename Block_>
153std::vector<SizeFactor_> compute_blocked_mean(size_t num, const SizeFactor_* size_factors, const Block_* block, sanitize_size_factors::Diagnostics* diagnostics, const Options& options) {
154 static_assert(std::is_floating_point<SizeFactor_>::value);
155 size_t ngroups = tatami_stats::total_groups(block, num);
156 std::vector<SizeFactor_> group_mean(ngroups);
157 std::vector<size_t> group_num(ngroups);
158
159 if (options.ignore_invalid) {
160 sanitize_size_factors::Diagnostics tmpdiag;
161 auto& diag = (diagnostics == NULL ? tmpdiag : *diagnostics);
162 for (size_t i = 0; i < num; ++i) {
163 auto val = size_factors[i];
164 if (!sanitize_size_factors::internal::is_invalid(val, diag)) {
165 auto b = block[i];
166 group_mean[b] += val;
167 ++(group_num[b]);
168 }
169 }
170 } else {
171 for (size_t i = 0; i < num; ++i) {
172 auto b = block[i];
173 group_mean[b] += size_factors[i];
174 ++(group_num[b]);
175 }
176 }
177
178 for (size_t g = 0; g < ngroups; ++g) {
179 if (group_num[g]) {
180 group_mean[g] /= group_num[g];
181 }
182 }
183
184 return group_mean;

203template<typename SizeFactor_, typename Block_>
204std::vector<SizeFactor_> compute_blocked(size_t num, SizeFactor_* size_factors, const Block_* block, sanitize_size_factors::Diagnostics* diagnostics, const Options& options) {
205 auto group_mean = compute_blocked_mean(num, size_factors, block, diagnostics, options);
206
207 if (options.block_mode == BlockMode::PER_BLOCK) {
208 for (size_t i = 0; i < num; ++i) {
209 const auto& div = group_mean[block[i]];
210 if (div) {
211 size_factors[i] /= div;
212 }
213 }
214
215 } else if (options.block_mode == BlockMode::LOWEST) {
216 SizeFactor_ min = 0;
217 bool found = false;
218 for (auto m : group_mean) {
219 // Ignore groups with means of zeros, either because they're full
220 // of zeros themselves or they have no cells associated with them.
221 if (m) {
222 if (!found || m < min) {
223 min = m;
224 found = true;
225 }
226 }
227 }
228
229 if (min > 0) {
230 for (size_t i = 0; i < num; ++i) {
231 size_factors[i] /= min;
232 }
233 }
234 }

236 return group_mean;
